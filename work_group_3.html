def connect_mysql(host, user, password, database):
    try:
        connection = pymysql.connect(host=host,
                                     user=user,
                                     password=password,
                                     database=database,
                                     charset='utf8mb4',
                                     cursorclass=pymysql.cursors.DictCursor)
        
        print("")
        return connection
    
    except pymysql.MySQLError as e:
        print(f"{e}")
        return None

def check(connection):
    if connection is None:
        print("")
        return
    
    try:
        with connection.cursor() as cursor:
            sql = "SELECT * FROM student"

            cursor.execute(sql)
            
            rows = cursor.fetchall()

            if rows:
                for row in rows:
                    print(row)

            else:
                print("")

    except pymysql.MySQLError as e:
        print(f"{e}")

def add_data(connection, new_data):
    if connection is None:
        print("")
        return
    
    try:
        with connection.cursor() as cursor:
            sql = "INSERT INTO student (id, name, sex, score) VALUES (%s, %s, %s, %s)"

            cursor.execute(sql, (new_data['id'], new_data['name'], new_data['sex'],new_data['score']))
            connection.commit()

            print("")

    except pymysql.MySQLError as e:
        print(f"{e}")
